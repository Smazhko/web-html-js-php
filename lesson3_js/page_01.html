=============== ТОТ ЖЕ КОД, НО С ИСПОЛЬЗОВАНИЕМ SWITCH ==============
let age = 30;

switch (age){
    case 10:                       
        alert("Пора учить уроки, завтра контрольная!");
        break;
    case 18:
        alert("Вы совершеннолетний. Можно всё.");
        break;
    case 30:
        alert("Пора в люльку, завтра рано вставать на работу!");
        break;
    default:
        alert("ИИ ещё не вычислил направление Вашей судьбы...");
}
